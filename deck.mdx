```jsx
export const InputWithLabel = ({ label, id, onInputChange, ...rest }) => (
  <FormGroup>
    <Label htmlFor={id}>{label}:</Label>
    <Input
      id={id}
      name={id}
      onChange={e => onInputChange && onInputChange(id, e.target.value)}
      {...rest}
    />
  </FormGroup>
);
```

```jsx
  <Form {...rest}>
    <InputWithLabel
      id="email"
      label="Email"
      value={(values || {}).email}
      onInputChange={onInputChange}
    />
    <InputWithLabel
      id="passowrd"
      label="Password"
      type="passowrd"

    />

);
```

# Pages

|   |   |
|---|---|
| `<LoginPage />` | ...  |

```jsx
export class LoginPage extends Component {
  state = {
    email: '',
    passowrd: '',
  };

  render() {
    const { email, passowrd } = this.state;
    const values = { email, passowrd };

    return (
      <LoginTemplate
        values={values}
        onInputChange={(id, value) =>
          this.setState({ [id]: value })
        }
        onSubmit={e => {
          e.preventDefault();
          alert(`Send data: ${JSON.stringify(values)}`);
        }}
      />
    );
  }
}
```
